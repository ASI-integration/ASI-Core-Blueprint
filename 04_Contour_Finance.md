Базовые объекты контура

2. Money State
Состояние денег в системе:

EXPECTED — ожидаются

RECEIVED — получены

LOCKED — нельзя трогать (проживание не завершено)

EARNED — заработаны системой

REFUNDABLE — допустим возврат

FORFEITED — удержаны

Важно: Money State ≠ Booking State
Они связаны, но не равны (как и с объектом в Contour 3).

Каноны контура (черновик)

До исхода проживания деньги находятся в состоянии LOCKED

Возврат — это следствие исхода, а не желания гостя

Finance не «решает» — он фиксирует допустимость

Связь с предыдущими контурами

import fetch from "node-fetch";

const TOKEN = process.env.TELEGRAM_BOT_TOKEN;
const API_URL = `https://api.telegram.org/bot${TOKEN}`;

let offset = 0;

async function poll() {
  const url = `${API_URL}/getUpdates?timeout=30&offset=${offset}`;

  const res = await fetch(url);
  const data = await res.json();

  if (!data.ok) {
    console.error("Telegram error:", data);
    return;
  }

  for (const update of data.result) {
    offset = update.update_id + 1;

    if (update.message?.text === "/start") {
      const chatId = update.message.chat.id;

      await fetch(`${API_URL}/sendMessage`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          chat_id: chatId,
          text: "Привет! Бот жив 👋",
        }),
      });

      console.log("/start обработан");
    }
  }
}

async function startPolling() {
  while (true) {
    try {
      await poll();
    } catch (e) {
      console.error("Polling error:", e);
    }
  }
}

📄 app/telegram/bot.js — АГЕНТ СО СОСТОЯНИЯМИ (ГОТОВЫЙ)
👉 Полностью замени файл app/telegram/bot.js этим содержимым:

const TOKEN = "8088296273:AAEHjsbfo1BelVmvNulALfp7T1bSUHVsk60";
const API_URL = `https://api.telegram.org/bot${TOKEN}`;

let offset = 0;

// ===== СОСТОЯНИЯ =====
const State = {
  IDLE: "idle",
  WAITING_GOAL: "waiting_goal",
};

const stateByChat = {};

// ===== API HELPERS =====
async function api(method, payload) {
  const res = await fetch(`${API_URL}/${method}`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(payload),
  });
  return res.json();
}

async function send(chatId, text, keyboard) {
  return api("sendMessage", {
    chat_id: chatId,
    text,
    reply_markup: keyboard,
  });
}

// ===== POLLING =====
async function poll() {
  const res = await fetch(
    `${API_URL}/getUpdates?timeout=30&offset=${offset}`
  );
  const data = await res.json();

  if (!data.ok) {
    console.error("Telegram error:", data);
    return;
  }

  for (const update of data.result) {
    offset = update.update_id + 1;

    // ===== /start =====
    if (update.message?.text === "/start") {
      const chatId = update.message.chat.id;

      stateByChat[chatId] = State.IDLE;

      await send(
        chatId,
        "Привет! Я агент StayAutomated 👋\nВыбери действие:",
        {
          inline_keyboard: [
            [{ text: "🚀 Начать сценарий", callback_data: "start_flow" }],
            [{ text: "ℹ️ Что я умею", callback_data: "about" }],
          ],
        }
      );

      console.log("/start → state = IDLE");
    }

    // ===== КНОПКИ =====
    if (update.callback_query) {
      const chatId = update.callback_query.message.chat.id;
      const action = update.callback_query.data;

      if (action === "about") {
        await send(
          chatId,
          "Я агент с памятью состояния.\n" +
            "Я умею:\n" +
            "• вести диалог по шагам\n" +
            "• задавать вопросы\n" +
            "• реагировать по контексту 🤖"
        );
      }

      if (action === "start_flow") {
        stateByChat[chatId] = State.WAITING_GOAL;

        await send(
          chatId,
          "Отлично 👍\nНапиши, что ты хочешь автоматизировать?"
        );

        console.log("state → WAITING_GOAL");
      }

      await api("answerCallbackQuery", {
        callback_query_id: update.callback_query.id,
      });
    }

    // ===== ТЕКСТ ПО СОСТОЯНИЮ =====
    if (update.message?.text && update.message.text !== "/start") {
      const chatId = update.message.chat.id;
      const text = update.message.text;
      const state = stateByChat[chatId];

      if (state === State.WAITING_GOAL) {
        await send(
          chatId,
          `Принято 👍\nТы хочешь автоматизировать:\n"${text}"\n\n` +
            "Следующий шаг — скоро 🚧"
        );

        stateByChat[chatId] = State.IDLE;
        console.log("goal captured → back to IDLE");
      }
    }
  }
}

// ===== LOOP =====
async function start() {
  while (true) {
    try {
      await poll();
    } catch (e) {
      console.error("Polling error:", e);
    }
  }
}

start();

  });

// StayAutomated — Telegram Bot (stable version)

import dotenv from "dotenv";
import path from "path";
import { fileURLToPath } from "url";
import fetch from "node-fetch";

/* =======================
   ENV SETUP (NO MAGIC)
   ======================= */
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

// грузим .env.local из корня проекта
dotenv.config({
  path: path.resolve(__dirname, "../.env.local"),
});

/* =======================
   TOKEN
   ======================= */
const TOKEN = process.env.TELEGRAM_BOT_TOKEN;

if (!TOKEN) {
  console.error("❌ TELEGRAM_BOT_TOKEN is not defined in .env.local");
  process.exit(1);
}

console.log("✅ Telegram token loaded");

/* =======================
   TELEGRAM API
   ======================= */
const API = `https://api.telegram.org/bot${TOKEN}`;
let offset = 0;

/* =======================
   HELPERS
   ======================= */
async function sendMessage(chatId, text) {
  await fetch(`${API}/sendMessage`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      chat_id: chatId,
      text,
    }),
  });
}

/* =======================
   UPDATE HANDLER
   ======================= */
async function handleUpdate(update) {
  if (!update.message) return;

  const chatId = update.message.chat.id;
  const text = update.message.text || "";

  if (text === "/start") {
    await sendMessage(chatId, "🤖 StayAutomated bot is alive.");
  } else {
    await sendMessage(chatId, `Ты написал: ${text}`);
  }
}

/* =======================
   POLLING LOOP
   ======================= */
async function poll() {
  try {
    const res = await fetch(
      `${API}/getUpdates?timeout=30&offset=${offset}`
    );
    const data = await res.json();

    if (data.result) {
      for (const update of data.result) {
        offset = update.update_id + 1;
        await handleUpdate(update);
      }
    }
  } catch (err) {
    console.error("Polling error:", err.message);
  }

  setTimeout(poll, 1000);
}

/* =======================
   START
   ======================= */
console.log("🚀 Telegram bot started");
poll();

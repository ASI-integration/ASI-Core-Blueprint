1️⃣ Базовые якоря (без них ORL не считается)
Если хотя бы один пункт неизвестен или противоречив → ORL = UNKNOWN:

неизвестен Arrival Outcome

неизвестно, был ли выдан доступ

текущее время не соотносится с окном проживания

есть ручной флаг «сомнение»

UNKNOWN — это не ошибка. Это честность системы.

2️⃣ Канон LOW RISK 🟢
ORL = LOW, если одновременно:

Arrival Outcome = CONFIRMED_STAY

доступ выдан и использован в допустимое время

текущее время внутри периода проживания

нет ручных флагов

Интуитивно: гость живёт, всё ок, не дёргаемся.

3️⃣ Канон MEDIUM RISK 🟡
ORL = MEDIUM, если:

проживание возможно, но не подтверждено однозначно
или

есть расхождения без прямой угрозы

Типовые примеры:

доступ выдан, но не использован

Arrival Outcome = UNDETERMINED

время вышло, но нет подтверждённого выезда

ранний выезд заявлен, но не верифицирован

Это зона «наблюдаем, но не паникуем».

4️⃣ Канон HIGH RISK 🔴
ORL = HIGH, если есть прямые противоречия:

доступ использован вне допустимого окна

Arrival Outcome = NO_SHOW, но доступ использовался

бронь отменена, но признаки проживания есть

конфликт сигналов + отсутствие ручного разрешения

HIGH = потенциальный инцидент.
Но не автоматическое действие — только сигнал.

5️⃣ Роль ручного override (очень важно)
Ручной флаг оператора / владельца:

может повысить риск (LOW → MEDIUM / HIGH)

может зафиксировать UNKNOWN осознанно

не может понизить HIGH без устранения противоречий

Человеческий здравый смысл > формальная логика.
Но только через явную фиксацию.

// ASI_CORE: OwnerPreferences.ts
// Owner-defined contractors & fallback rules for Security / Emergency / Ops

export type ContractorRole =
  | "housekeeping"
  | "plumber"
  | "electrician"
  | "security"
  | "insurance"
  | "building_management";

export type OwnerContractor = {
  role: ContractorRole;
  name: string;
  phone?: string;
  email?: string;
  notes?: string;
  priority?: number; // lower = higher priority
};

export type FallbackRule = {
  role: ContractorRole;
  timeoutMin: number; // wait before fallback
  fallback: "marketplace" | "partner" | "manual";
};

export type OwnerPreferences = {
  propertyId: string;

  contractors: OwnerContractor[];

  fallbackRules: FallbackRule[];

  nightPolicy?: "allow_emergency_only" | "allow_all" | "manual_only";
};

// --------- helpers ---------

export function getContractorsForRole(
  prefs: OwnerPreferences,
  role: ContractorRole
): OwnerContractor[] {
  return prefs.contractors
    .filter((c) => c.role === role)
    .sort((a, b) => (a.priority ?? 100) - (b.priority ?? 100));
}

export function getFallbackRule(
  prefs: OwnerPreferences,
  role: ContractorRole
): FallbackRule | undefined {
  return prefs.fallbackRules.find((r) => r.role === role);
}

"use client";

import React from "react";
import { useAuditStore } from "./AuditStore";
import type { DecisionAuditRecord } from "./DecisionAudit";

type Props = {
  propertyId: string;
};

function OutcomeBadge({ outcome }: { outcome: DecisionAuditRecord["result"]["outcome"] }) {
  const map: Record<string, string> = {
    auto_executed: "bg-green-100 text-green-800",
    assisted_required: "bg-yellow-100 text-yellow-800",
    manual_only: "bg-slate-100 text-slate-700",
  };

  return (
    <span className={`rounded-full px-2 py-0.5 text-xs font-semibold ${map[outcome]}`}>
      {outcome.replace("_", " ")}
    </span>
  );
}

export default function AuditTimelinePanel({ propertyId }: Props) {
  const timeline = useAuditStore((s) => s.getTimeline(propertyId));

  if (!timeline.length) {
    return (
      <div className="rounded-2xl border bg-white p-4 text-sm text-slate-500">
        Audit timeline пока пуст.
      </div>
    );
  }

  return (
    <div className="flex flex-col gap-4">
      {timeline.map((audit) => (
        <div key={audit.auditId} className="rounded-2xl border bg-white p-4 shadow-sm">
          <div className="mb-2 flex items-center justify-between">
            <div className="text-sm font-semibold text-slate-900">
              Событие: {audit.event.type}
            </div>
            <OutcomeBadge outcome={audit.result.outcome} />
          </div>

          <div className="mb-3 text-xs text-slate-600">
            Confidence: {audit.event.confidence} · Confirmations:{" "}
            {audit.event.confirmations} · Mode: {audit.context.securityMode}
          </div>

          <div className="mb-3 text-xs text-slate-600">
            Object: {audit.context.objectType}
            {audit.context.playbookId && ` · Playbook: ${audit.context.playbookId}`}
          </div>

          <div className="flex flex-col gap-2">
            {audit.decisions.map((d, i) => (
              <div key={i} className="rounded-lg bg-slate-50 p-2 text-xs">
                <div className="font-semibold text-slate-700">{d.step}</div>
                <div className="text-slate-600">{d.explanation}</div>
              </div>
            ))}
          </div>
        </div>
      ))}
    </div>
  );
}

// ASI_CORE: AuditShareResolver.ts
// Real resolver for share-audit tokens (in-memory implementation)
//
// This is production-safe in terms of logic.
// Storage can later be swapped to DB / KV without changing API.

import type { DecisionAuditRecord } from "./DecisionAudit";
import type { AuditShareToken } from "./AuditShareToken";
import { isTokenValid } from "./AuditShareToken";

// --- In-memory stores (replace later with DB / KV) ---

const tokenStore = new Map<string, AuditShareToken>();
const auditStore = new Map<string, DecisionAuditRecord>();

// --- Registration API ---

export function registerAuditForSharing(
  audit: DecisionAuditRecord,
  token: AuditShareToken
) {
  tokenStore.set(token.token, token);
  auditStore.set(audit.auditId, audit);
}

// --- Resolver ---

export function resolveAuditByToken(tokenValue: string): {
  audit: DecisionAuditRecord | null;
  meta: AuditShareToken | null;
} {
  const meta = tokenStore.get(tokenValue);
  if (!meta) return { audit: null, meta: null };
  if (!isTokenValid(meta)) return { audit: null, meta };

  const audit = auditStore.get(meta.auditId) ?? null;
  return { audit, meta };
}

// ASI_CORE: CorporateFitEvaluator.ts
// Evaluates Corporate Fit based on Object Context

import type { ObjectContext } from "./ContextTypes";

export type CorporateFitResult = {
  level: "low" | "medium" | "high";
  reasons: string[];
};

export function evaluateCorporateFit(
  context: ObjectContext
): CorporateFitResult {
  const reasons: string[] = [];

  if (context.location.businessDensity === "high") {
    reasons.push("Высокая концентрация бизнес-активности");
  }

  if (context.location.transportAccessibility === "high") {
    reasons.push("Хорошая транспортная доступность");
  }

  if (context.demand.weekdayDemand === "high") {
    reasons.push("Сильный спрос в будние дни");
  }

  if (context.usage.recommendedStayLength === "long") {
    reasons.push("Подходит для длительных заездов");
  }

  const score = reasons.length;

  if (score >= 3) return { level: "high", reasons };
  if (score === 2) return { level: "medium", reasons };
  return { level: "low", reasons };
}

// ASI_CORE: ContextSummary.ts
// Human-readable context explanation for UI / Audit

import type { ObjectContext } from "./ContextTypes";
import { evaluateCorporateFit } from "./CorporateFitEvaluator";

export function summarizeContext(context: ObjectContext): string[] {
  const summary: string[] = [];

  if (!context.location.isPrimeLocation) {
    summary.push("Объект расположен вне туристической prime-локации");
  }

  const corporate = evaluateCorporateFit(context);

  if (corporate.level === "high") {
    summary.push("Объект хорошо подходит для корпоративных заездов");
  }

  if (context.demand.seasonality === "high") {
    summary.push("Выраженная сезонность спроса");
  }

  if (context.demand.weekdayDemand === "high") {
    summary.push("Стабильный спрос в будние дни");
  }

  return summary;
}

import type { StrategyMode } from "./StrategySwitchAdvisor";
import type { StrategySwitchAdvice } from "./StrategySwitchAdvisor";
import type { DecisionAuditRecord } from "../audit/DecisionAudit";

export function buildStrategyRecommendationAudit(
  propertyId: string,
  advice: StrategySwitchAdvice
): DecisionAuditRecord {
  return {
    auditId: crypto.randomUUID(),
    propertyId,
    createdAt: Date.now(),
    event: {
      type: "strategy_recommendation",
      severity: "info",
      confidence: 0.7,
      confirmations: 0,
    },
    context: {
      objectType: "property",
      securityMode: "n/a",
    },
    decisions: advice.reasons.map((r) => ({
      step: "recommendation_reason",
      explanation: r,
    })),
    result: {
      outcome: "assisted_required",
      note: `Рекомендована стратегия: ${advice.to}`,
    },
  };
}

export function buildStrategyAcceptedAudit(
  propertyId: string,
  from: StrategyMode,
  to: StrategyMode
): DecisionAuditRecord {
  return {
    auditId: crypto.randomUUID(),
    propertyId,
    createdAt: Date.now(),
    event: {
      type: "strategy_accepted",
      severity: "info",
      confidence: 1,
      confirmations: 1,
    },
    context: {
      objectType: "property",
      securityMode: "n/a",
    },
    decisions: [
      {
        step: "user_action",
        explanation: `Пользователь принял смену стратегии с ${from} на ${to}`,
      },
    ],
    result: {
      outcome: "manual_only",
    },
  };
}
